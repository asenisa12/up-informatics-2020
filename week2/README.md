#### **БИНАРНИ  ОПЕРАТОРИ**
```cpp

int main() {
    int v1 = 5, v2 = 12; // 0101, 1100 

	// ## v4
	// binary operations
	// & (and):  0101 & 1100 = 0100
	// | (or) :  0101 | 1100 = 1101
	// << (shift left):  0101 << 1 = 1010
	// >> (shift right): 1010 >> 1 = 101
	// ^ (xor exclusive or): 0110 ^ 1010 = 1100

    //ПРИМЕРИ:
    int v3 = v1 & v2; // 0100 = 4
    int v4 = v1 | v2; // 1101 = 13
    int v5 = v1 << 1; // 1010 = 10
    int v6 = v1 ^ 1; // 1100 = 12
    
    return 0;
}
```
