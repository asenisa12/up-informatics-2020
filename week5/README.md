#### **Функции**
```cpp
#include <iostream> 
using namespace std;

void print_args(int arg1, float arg2, char arg3)
{
    cout<<"arg1: " << arg1 << ", arg2:" << arg2 << ", arg3:" << arg3 << endl
}

int max(int arg1, int arg2)
{
    if(arg1>arg2)
    {
        return arg1;
    }
    else
    {
        return arg2;
    }
}

int main() {
    print_args(1, 2.2, 'a');// prints "arg1: 1, arg2:2.2, arg3:a"
    cout<<max(1,2)<<endl; // prints "2"
    return 0;
}
```
